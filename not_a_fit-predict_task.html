We will use recall as main metric for CV as recall + precision pair has a nice business interpretation.

In [31]:
# Scale features for better convergence

scaler = StandardScaler()  

C_values = [0.001, 0.01, 0.05, 0.1]

param_grid = {'logisticregression__C': C_values, 
              'logisticregression__penalty': ['l1', 'l2']}

clf = LogisticRegression(solver='liblinear', random_state=RANDOM_STATE, multi_class='auto')

pipeline = make_pipeline(scaler, clf)

estimator = GridSearchCV(pipeline, param_grid=param_grid, scoring='recall_micro', cv=3)
estimator.fit(X_train, y_train)

print("Mean recall for the 1 class across the folds on cv: {0} \n".format(estimator.best_score_))
print("Best estimator: \n {0} \n".format(estimator.best_estimator_))
print("Classification report for holdout: \n", classification_report(y_test, 
                                                                    estimator.best_estimator_.predict(X_test)))
